$$x_1 = \dfrac{8c \pm \sqrt{80 - 16c^2}}{10}$$

### –ü—Ä–∏–º–µ—Ä 2

$\text{extr}\left(\left|x_1\right|^\lambda + 2 \left|x_2\right|^\lambda \right)$ –ø—Ä–∏ —É—Å–ª–æ–≤–∏–∏ $x_1^2 + 4x_2^2 \le 4, \;\; \lambda \neq 0$

–†–µ—à–µ–Ω–∏–µ:

* –≠—Ç–∞ –∑–∞–¥–∞—á–∞ - —Å –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–º $\lambda$, –ø–æ—ç—Ç–æ–º—É –∑–¥–µ—Å—å –æ—á–µ–Ω—å –≤–∞–∂–Ω–æ —Ä–∞—Å—Å–º–æ—Ç—Ä–µ—Ç—å –≤—Å–µ —Å–ª—É—á–∞–∏. –î–ª—è –Ω–∞—á–∞–ª–∞ –æ–≥—Ä–∞–Ω–∏—á–∏–º—Å—è –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–º–∏ –∑–Ω–∞—á–µ–Ω–∏—è–º–∏ $\lambda$. 
* –í–∑–≥–ª—è–Ω–∏—Ç–µ –Ω–∞ —Ä–∏—Å—É–Ω–∫–∏ –ø—Ä–∏ $\lambda < 1, \lambda = 1, \lambda > 1$.


```python
import plotly
import plotly.plotly as py
import plotly.graph_objs as go
from plotly import tools
plotly.offline.init_notebook_mode(connected=True)

import numpy as np

precise = 40
r = np.linspace(0,1,precise)                  # radius
phi = np.linspace(0, 2*np.pi, precise)        # angle

r, phi = np.meshgrid(r, phi)              # computational grid

x1 = 2 * r * np.cos(phi)                         # parametrization
x2 = r * np.sin(phi)
z = np.zeros((precise,precise))

budget_set = go.Surface(x=x1, y=x2, z=z, opacity = 0.9)
lam = [0.5, 1, 2]
fig = tools.make_subplots(rows=1, cols=len(lam),specs=[[{'is_3d': True} for i in range(len(lam))]])
f_ = {}
for lamb in lam:
    f_[lamb] = np.abs(x1)**lamb + 2*np.abs(x2)**lamb         # target function

target_function = {}

data = [budget_set]

fig['layout'].update(title='lambda > 0')
scene = dict(
    xaxis=dict(
        title = 'x1',
        gridcolor='rgb(255, 255, 255)',
        zerolinecolor='rgb(255, 255, 255)',
        showbackground=True,
        backgroundcolor='rgb(230, 230,230)'
    ),
    yaxis=dict(
        title = 'x2',
        gridcolor='rgb(255, 255, 255)',
        zerolinecolor='rgb(255, 255, 255)',
        showbackground=True,
        backgroundcolor='rgb(230, 230,230)'
    ),
    zaxis=dict(
        title = 'f(x1, x2)',
        gridcolor='rgb(255, 255, 255)',
        zerolinecolor='rgb(255, 255, 255)',
        showbackground=True,
        backgroundcolor='rgb(230, 230,230)'
    )
)
i = 1
for lamb in lam:
    target_function[lamb] = go.Surface(x = x1, y = x2, z=f_[lamb], opacity = 0.8, name = 'lambda = '+str(lamb))
    data.append(target_function[lamb])
    fig.append_trace(budget_set, 1, i)
    fig.append_trace(target_function[lamb], 1, i)
    fig['layout']['scene'+str(i)].update(scene)
    i += 1

plotly.offline.iplot(fig)
```

–ü–æ —Ö–æ—Ä–æ—à–µ–º—É, –∑–∞–¥–∞—á–∞ —Ä–µ—à–∞–µ—Ç—Å—è –∞–±—Å–æ–ª—é—Ç–Ω–æ —Ç–∞–∫ –∂–µ - —Ä–∏—Å–æ–≤–∞–Ω–∏–µ–º –ª–∏–Ω–∏–π —É—Ä–æ–≤–Ω–µ–π $f(x_1, x_2) = const$ –∏ –ø—Ä–∏—Ä–∞–≤–Ω–∏–≤–∞–Ω–∏–µ–º —É—Ä–∞–≤–Ω–µ–Ω–∏–π –∫–∞—Å–∞—Ç–µ–ª—å–Ω—ã—Ö. –û–¥–Ω–∞–∫–æ, –≤ —ç—Ç–æ–π –∑–∞–¥–∞—á–µ —è—Å–Ω–æ, —á—Ç–æ $(0, 0)$ - —Ç–æ—á–∫–∞ $min$ (–æ–Ω–∞ –∂–µ –∏–Ω—Ñ–∏–º—É–º —ç—Ç–æ–π –∑–∞–¥–∞—á–∏). –ß—Ç–æ –∂–µ –∫–∞—Å–∞–µ—Ç—Å—è –º–∞–∫—Å–∏–º—É–º–æ–≤, –¥–∞–≤–∞–π—Ç–µ –ø–æ—Å–º–æ—Ç—Ä–∏–º:

$$
 \begin{cases}
   x_1^2 + 4x_2^2 = 4, \\
   \left|x_1\right|^\lambda + 2 \left|x_2\right|^\lambda = c
 \end{cases}
$$

–í —Å–∏–ª—É —Å–∏–º–º–µ—Ç—Ä–∏–∏, –º–æ–∂–µ–º —Ä–∞—Å—Å–º–æ—Ç—Ä–µ—Ç—å –ª–∏—à—å –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è $x_1, x_2$:

$$
 \begin{cases}
   x_1^2 + 4x_2^2 = 4, \\
   x_1^\lambda + 2 x_2^\lambda = c
 \end{cases}
$$

–í–æ–æ–±—â–µ –≥–æ–≤–æ—Ä—è, –Ω–∞–¥–æ —Ä–µ—à–∏—Ç—å —ç—Ç—É —Å–∏—Å—Ç–µ–º—É –¥–ª—è –≤—Å–µ—Ö $\lambda$ –∏ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è $c$, —Ç.–µ. –º–æ–∂–Ω–æ –≤—ã—Ä–∞–∑–∏—Ç—å $x_2 = x_2(x_1)$ –∏ –ø—Ä–∏—Ä–∞–≤–Ω—è—Ç—å –ø—Ä–æ–∏–∑–≤–æ–¥–Ω—ã–µ –ø–æ $x_1$ –ø–µ—Ä–≤–æ–≥–æ –∏ –≤—Ç–æ—Ä–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏–π (—É—Å–ª–æ–≤–∏–µ –∫–∞—Å–∞–Ω–∏—è). –û—Å–æ–±—ã–π –∏–Ω—Ç–µ—Ä–µ—Å –≤—ã–∑—ã–≤–∞–µ—Ç –ø–æ–≤–µ–¥–µ–Ω–∏–µ —Ä–µ—à–µ–Ω–∏–π –ø—Ä–∏ $\lambda \in (1,2)$, –ø—Ä–∏ –≤—Å–µ—Ö –æ—Å—Ç–∞–ª—å–Ω—ã—Ö –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã—Ö –∑–Ω–∞—á–µ–Ω–∏—è—Ö —Ä–µ—à–µ–Ω–∏—è —Å–æ–≤–ø–∞–¥–∞—é—Ç —Å —Ç–µ–º–∏, —á—Ç–æ —É–∫–∞–∑–∞–Ω—ã –Ω–∞ —Ä–∏—Å—É–Ω–∫–µ –≤—ã—à–µ. –ß–∏—Ç–∞—Ç–µ–ª—é –ø—Ä–µ–¥–ª–∞–≥–∞–µ—Ç—Å—è –ø—Ä–∏–¥—É–º–∞—Ç—å —Å–ø–æ—Å–æ–± –Ω–∞–π—Ç–∏ –∞–Ω–∞–ª–∏—Ç–∏—á–µ—Å–∫–æ–µ —Ä–µ—à–µ–Ω–∏–µ –≤ —ç—Ç–æ–º —Å–ª—É—á–∞–µ.

–í —Å–ª—É—á–∞–µ –∂–µ $\lambda <0$ —Ç–æ—á–∫–∞ $(0,0)$ —è–≤–ª—è–µ—Ç—Å—è —Å—É–ø—Ä–µ–º—É–º–æ–º —Ü–µ–ª–µ–≤–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ –Ω–∞ –∑–∞–¥–∞–Ω–Ω–æ–º –º–Ω–æ–∂–µ—Å—Ç–≤–µ. –ò —ç—Ç—É —Ç–æ—á–∫—É —Å—Ç–æ–∏—Ç –≤–∫–ª—é—á–∏—Ç—å –≤ –æ—Ç–≤–µ—Ç. –ì—Ä–∞—Ñ–∏–∫–∏ –Ω–∏–∂–µ.


```python
import plotly
import plotly.plotly as py
import plotly.graph_objs as go
from plotly import tools
plotly.offline.init_notebook_mode(connected=True)

import numpy as np

precise = 40
r = np.linspace(0,1,precise)                  # radius
phi = np.linspace(0, 2*np.pi, precise)        # angle

r, phi = np.meshgrid(r, phi)              # computational grid

x1 = 2 * r * np.cos(phi)                         # parametrization
x2 = r * np.sin(phi)
z = np.zeros((precise,precise))

budget_set = go.Surface(x=x1, y=x2, z=z, opacity = 0.9)
lam = [-0.5, -1, -2]
fig = tools.make_subplots(rows=1, cols=len(lam),specs=[[{'is_3d': True} for i in range(len(lam))]])
f_ = {}
for lamb in lam:
    f_[lamb] = np.abs(x1)**lamb + 2*np.abs(x2)**lamb         # target function

target_function = {}

data = [budget_set]

fig['layout'].update(title='lambda < 0')
scene = dict(
    xaxis=dict(
        title = 'x1',
        gridcolor='rgb(255, 255, 255)',
        zerolinecolor='rgb(255, 255, 255)',
        showbackground=True,
        backgroundcolor='rgb(230, 230,230)'
    ),
    yaxis=dict(
        title = 'x2',
        gridcolor='rgb(255, 255, 255)',
        zerolinecolor='rgb(255, 255, 255)',
        showbackground=True,
        backgroundcolor='rgb(230, 230,230)'
    ),
    zaxis=dict(
        title = 'f(x1, x2)',
        gridcolor='rgb(255, 255, 255)',
        zerolinecolor='rgb(255, 255, 255)',
        showbackground=True,
        backgroundcolor='rgb(230, 230,230)',
        range = [0,20]
    )
)
i = 1
for lamb in lam:
    target_function[lamb] = go.Surface(x = x1, y = x2, z=f_[lamb], opacity = 0.8, name = 'lambda = '+str(lamb))
    data.append(target_function[lamb])
    fig.append_trace(budget_set, 1, i)
    fig.append_trace(target_function[lamb], 1, i)
    fig['layout']['scene'+str(i)].update(scene)
    i += 1

plotly.offline.iplot(fig)
```
